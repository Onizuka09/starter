11 // Helper to implement the "base-from-member" idiom

23  void deallocate(void* ptr) {

109  // Reduces the capacity of the memory pool to match the current usage.
110  // https://arduinojson.org/v6/api/basicjsondocument/shrinktofit/
111  void shrinkToFit() {
112  ptrdiff_t bytes_reclaimed = _pool.squash();
113  if (bytes_reclaimed == 0)
114  return;
115 
116  void* old_ptr = _pool.buffer();
117  void* new_ptr = this->reallocate(old_ptr, _pool.capacity());
118 
119  ptrdiff_t ptr_offset =
120  static_cast<char*>(new_ptr) - static_cast<char*>(old_ptr);
121 
122  _pool.movePointers(ptr_offset);
123  _data.movePointers(ptr_offset, ptr_offset - bytes_reclaimed);
124  }

126  // Reclaims the memory leaked when removing and replacing values.
127  // https://arduinojson.org/v6/api/jsondocument/garbagecollect/
128  bool garbageCollect() {
129  // make a temporary clone and move assign
130  BasicJsonDocument tmp(*this);
131  if (!tmp.capacity())
132  return false;
133  tmp.set(*this);
134  moveAssignFrom(tmp);
135  return true;
136  }

141  MemoryPool allocPool(size_t requiredSize) {
142  size_t capa = addPadding(requiredSize);
143  return MemoryPool(reinterpret_cast<char*>(this->allocate(capa)), capa);
144  }

146  void reallocPool(size_t requiredSize) {
147  size_t capa = addPadding(requiredSize);
148  if (capa == _pool.capacity())
149  return;
150  freePool();

152  }

163  void moveAssignFrom(BasicJsonDocument& src) {
164  freePool();

166  _pool = src._pool;
