38 const uint8_t DS1307_REG_STATUS = 0x00;

40 const uint8_t DS1307_REG_RAMSTART = 0x08;
41 const uint8_t DS1307_REG_RAMEND = 0x3f;
42 const uint8_t DS1307_REG_RAMSIZE = DS1307_REG_RAMEND - DS1307_REG_RAMSTART;

117  void SetDateTime(const RtcDateTime& dt)
118  {
119  // retain running state
120  uint8_t sreg = getReg(DS1307_REG_STATUS) & _BV(DS1307_CH);
121 
122  // set the date time
123  _wire.beginTransmission(DS1307_ADDRESS);
124  _wire.write(DS1307_REG_TIMEDATE);
125 
126  _wire.write(Uint8ToBcd(dt.Second()) | sreg);
127  _wire.write(Uint8ToBcd(dt.Minute()));
128  _wire.write(Uint8ToBcd(dt.Hour())); // 24 hour mode only
129 
130  // RTC Hardware Day of Week is 1-7, 1 = Monday
131  // convert our Day of Week to Rtc Day of Week
132  uint8_t rtcDow = RtcDateTime::ConvertDowToRtc(dt.DayOfWeek());
133 
134  _wire.write(Uint8ToBcd(rtcDow));
135  _wire.write(Uint8ToBcd(dt.Day()));
136  _wire.write(Uint8ToBcd(dt.Month()));
137  _wire.write(Uint8ToBcd(dt.Year() - 2000));
138 
139  _lastError = _wire.endTransmission();
140  }

142  RtcDateTime GetDateTime()
143  {
144  _wire.beginTransmission(DS1307_ADDRESS);
145  _wire.write(DS1307_REG_TIMEDATE);
146  _lastError = _wire.endTransmission();
147  if (_lastError != Rtc_Wire_Error_None)
148  {
149  RtcDateTime(0);
150  }
151 
152  size_t bytesRead = _wire.requestFrom(DS1307_ADDRESS, DS1307_REG_TIMEDATE_SIZE);
153  if (DS1307_REG_TIMEDATE_SIZE != bytesRead)
154  {
155  _lastError = Rtc_Wire_Error_Unspecific;
156  RtcDateTime(0);
157  }
158 
159  uint8_t second = BcdToUint8(_wire.read() & 0x7F);
160  uint8_t minute = BcdToUint8(_wire.read());
161  uint8_t hour = BcdToBin24Hour(_wire.read());
162 
163  _wire.read(); // throwing away day of week as we calculate it
164 
165  uint8_t dayOfMonth = BcdToUint8(_wire.read());
166  uint8_t month = BcdToUint8(_wire.read());
167  uint16_t year = BcdToUint8(_wire.read()) + 2000;
168 
169  return RtcDateTime(year, month, dayOfMonth, hour, minute, second);
170  }

192  uint8_t SetMemory(uint8_t memoryAddress, const uint8_t* pValue, uint8_t countBytes)
193  {
194  uint8_t address = memoryAddress + DS1307_REG_RAMSTART;
195  uint8_t countWritten = 0;
196  if (address <= DS1307_REG_RAMEND)
197  {
198  _wire.beginTransmission(DS1307_ADDRESS);
199  _wire.write(address);
200 
201  while (countBytes > 0 && address <= DS1307_REG_RAMEND)
202  {
203  _wire.write(*pValue++);
204  address++;
205  countBytes--;
206  countWritten++;
207  }
208 
209  _lastError = _wire.endTransmission();
210  }
211  return countWritten;
212  }

214  size_t GetMemory(uint8_t memoryAddress, uint8_t* pValue, size_t countBytes)
215  {
216  uint8_t address = memoryAddress + DS1307_REG_RAMSTART;
217  size_t countRead = 0;
218  if (address <= DS1307_REG_RAMEND)
219  {
220  if (countBytes > DS1307_REG_RAMSIZE)
221  {
222  countBytes = DS1307_REG_RAMSIZE;
223  }
224 
225  _wire.beginTransmission(DS1307_ADDRESS);
226  _wire.write(address);
227  _lastError = _wire.endTransmission();
228  if (_lastError != Rtc_Wire_Error_None)
229  {
230  return 0;
231  }
232 
233  countRead = _wire.requestFrom(DS1307_ADDRESS, countBytes);
234  countBytes = countRead;
235 
236  while (countBytes-- > 0)
237  {
238  *pValue++ = _wire.read();
239  }
240  }
241 
242  return countRead;
243  }

254  uint8_t getReg(uint8_t regAddress)
255  {
256  _wire.beginTransmission(DS1307_ADDRESS);
257  _wire.write(regAddress);
258  _lastError = _wire.endTransmission();
259  if (_lastError != Rtc_Wire_Error_None)
260  {
261  return 0;
262  }
263 
264  // control register
265  size_t bytesRead = _wire.requestFrom(DS1307_ADDRESS, (size_t)1);
266  if (1 != bytesRead)

268  _lastError = Rtc_Wire_Error_Unspecific;
269  return 0;

271
