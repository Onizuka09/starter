30  * Platform IR input IR output Tone Core/Pin schema
31  * --------------------------------------------------------------
32  * DEFAULT/AVR 2 3 4 Arduino
33  * ATtinyX5 0|PB0 4|PB4 3|PB3 ATTinyCore
34  * ATtiny167 3|PA3 2|PA2 7|PA7 ATTinyCore
35  * ATtiny167 9|PA3 8|PA2 5|PA7 Digispark original core
36  * ATtiny84 |PB2 |PA4 |PA3 ATTinyCore
37  * ATtiny88 3|PD3 4|PD4 9|PB1 ATTinyCore
38  * ATtiny3217 18|PA1 19|PA2 20|PA3 MegaTinyCore
39  * ATtiny1604 2 3|PA5 %
40  * ATtiny816 14|PA1 16|PA3 1|PA5 MegaTinyCore
41  * ATtiny1614 8|PA1 10|PA3 1|PA5 MegaTinyCore
42  * SAMD21 3 4 5
43  * ESP8266 14|D5 12|D6 %
44  * ESP32 15 4 27
45  * ESP32-C3 2 3 4

203 }
204 void tone(uint8_t aPinNumber, unsigned int aFrequency, unsigned long aDuration){
205  ledcAttachPin(aPinNumber, TONE_LEDC_CHANNEL);
206  ledcWriteTone(TONE_LEDC_CHANNEL, aFrequency);
207  delay(aDuration);
208  ledcWriteTone(TONE_LEDC_CHANNEL, 0);
209 }
